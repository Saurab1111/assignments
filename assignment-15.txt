int duplicate(int a[],int size)
{
    int i,j,count=0,b[size],m=0,k=0;
    for(i=0;i<size;i++)
        for(j=i+1;j<size;j++)
    {
        if(a[i]==a[j]&&j<10)
        {
            for(k=0;k<m;k++)
            {
                if(j==b[k])
                    break;
            }
            if(k==m)
            {
                count++;
                b[m]=j;
                m++;
            }
        }
    }
    return count;
}

        }
